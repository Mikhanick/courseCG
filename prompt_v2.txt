
```text
Ты — эксперт по фреймворку UrbanSim3D. Создай **полностью готовую Python-функцию** для генерации 3D-модели здания. Функция должна:
1. Использовать классы и методы из `createBuilding.py`
2. Содержать подробные комментарии на русском языке в каждой секции
3. Детально описывать ВСЕ аргументы функций фреймворка
4. Сохранять результат в .json файл при запуске
5. единицы измерения должны быть эквивалентны метрам, например, высота блока в 2.8 подразумевает высоту 2.8 в реальном мире
6. Внутри кода при создании панелей исползуй координаты в натуральную величину, а не доли.

#### 🔑 СТРОГИЕ ТРЕБОВАНИЯ К ВЫВОДУ:
- **ТОЛЬКО КОД ФУНКЦИИ И ИМПОРТЫ** — никакого текста до/после кода
- Название функции: `generate_<название_здания>` (например, `generate_kiosk_pechat`)
- Обязательно включи `if __name__ == "__main__": generate_<название>()`
- Все комментарии — на русском языке, внутри кода
- Для КАЖДОГО вызова метода фреймворка опиши аргументы в комментариях

#### 📚 СТРУКТУРА КОДА (с обязательными комментариями):
```python
# Импорты — только необходимые из фреймворка
from createBuilding import (
    Building, GeometrySegment, RectangleWithCutout, SimplePolygon,
    RoofGeometry
)

def generate_<название_здания>():
    """
    Генерирует 3D-модель здания и сохраняет в .json файл.
    Описание здания: [краткое описание из запроса пользователя]
    """

    # === ШАГ 1: ИНИЦИАЛИЗАЦИЯ ЗДАНИЯ ===
    # Building.__init__ аргументы:
    #   name: str - уникальное имя модели (используется в метаданных)
    #   description: str - человекочитаемое описание здания
    #   author: str = "UrbanSim3D" - автор модели
    #   version: str = "1.0" - версия модели
    #   floor_count: int - общее количество этажей (включая цоколь)
    #   width_panels: int - количество ПАНЕЛЕЙ по ширине фасада
    #   depth_panels: int - количество ПАНЕЛЕЙ по глубине здания
    #   panel_width: float - ширина ОДНОЙ панели в метрах
    #   panel_depth: float - глубина ОДНОЙ панели в метрах
    #   floor_height: float - стандартная высота типового этажа
    #   ground_floor_height: Optional[float] - высота первого этажа (если None - равна floor_height)
    #   roof_type: str - тип крыши: "simple" (двускатная), "flat" (плоская с бортиком), "custom" (ручная настройка)
    #   roof_height: Optional[float] - высота крыши от основания (если None - 0.43 * floor_height)
    building = Building(
        name="<уникальное_имя_без_пробелов>",
        description="<подробное_описание_здания>",
        floor_count=1,          # Пример: 1 для киоска, 9 для панельного дома
        width_panels=3,         # Количество панелей по фасаду (X)
        depth_panels=2,         # Количество панелей по глубине (Z)
        panel_width=2.0,        # Ширина одной панели в метрах (X)
        panel_depth=2.0,        # Глубина одной панели в метрах (Z)
        floor_height=3.0,       # Высота типового этажа (Y)
        ground_floor_height=None,  # Если первый этаж выше/ниже типового
        roof_type="custom",     # "simple", "flat" или "custom"
        roof_height=1.5         # Высота крыши в метрах
    )

    # === ШАГ 2: СОЗДАНИЕ ГЕОМЕТРИЧЕСКИХ ПАНЕЛЕЙ ===
    # Building.create_window_frame() аргументы:
    #   margin: float (0.0-0.5) - отступ от краев панели до окна (в относительных координатах [0..1])
    #   frame_color: str - цвет рамки окна в формате #RRGGBB
    #   glass_color: Optional[str] - цвет стекла (если None - случайный темный цвет через get_window_color())
    #   invert_frame: bool - инвертировать нормали рамки (True для задних/боковых стен)
    #   invert_glass: bool - инвертировать нормали стекла
    front_panel = Building.create_window_frame(
        margin=0.15,                # 15% отступ от каждого края
        frame_color="#964B00",      # Коричневая рамка
        glass_color="#336699",      # Голубое стекло (явно задано)
        invert_frame=False,         # Нормали "наружу" для фасада
        invert_glass=False
    )

    # Building.create_solid_panel() аргументы:
    #   color: str - цвет панели в формате #RRGGBB
    #   invert: bool - инвертировать нормали (True для задних/боковых стен)
    back_panel = Building.create_solid_panel(
        color="#8B4513",            # Темно-коричневая глухая стена
        invert=True                 # Инверсия для задней стены (Z-направление)
    )

    # === ШАГ 3: ДОБАВЛЕНИЕ СТЕН ===
    # building.add_wall() аргументы:
    #   wall_type: str - тип стены: "front" (X+), "back" (X-), "left" (Z+), "right" (Z-)
    #   panels: List[GeometryComponent] - список панелей для этой стены
    #   floor_type: str - "ground" (первый этаж) или "typical" (этажи 2+)
    #   panels_count: Optional[int] - сколько панелей разместить (если None - len(panels))
    #   invert: bool - глобальная инверсия нормалей для ВСЕЙ стены
    # 
    # ВАЖНО: Для "back" и "right" стен обычно invert=True
    building.add_wall(
        wall_type="front",          # Передняя стена (Z=0)
        panels=[front_panel] * 3,   # 3 одинаковые панели окон
        floor_type="ground",        # Первый этаж
        panels_count=3,             # Явное указание количества
        invert=False                # Нормали смотрят наружу (+Z)
    )

    building.add_wall(
        wall_type="back",           # Задняя стена (Z=depth)
        panels=[back_panel] * 3,    # 3 глухие панели
        floor_type="ground",
        panels_count=3,
        invert=True                 # Инверсия для задней стены (-Z)
    )

    # building.complete_walls() аргументы:
    #   floor_type: str - "ground" или "typical"
    # Автоматически завершает недостающие стены:
    #   - Если есть front -> копирует на back (с инверсией)
    #   - Если есть left -> копирует на right (с инверсией)
    building.complete_walls("ground")  # Добавит left/right стены автоматически

    # === ШАГ 4: ТИПОВЫЕ ЭТАЖИ (если этажей > 1) ===
    if building.floor_count > 1 and building.typical_floor is not None:
        # Повторяем логику для типовых этажей (без специальных элементов вроде входа)
        typical_panel = Building.create_window_frame(
            margin=0.23,
            frame_color="#A4A4A4",
            glass_color=Building.get_window_color,  # Случайный темный цвет для окон
            invert_frame=False,
            invert_glass=False
        )
        
        building.add_wall("front", [typical_panel] * building.width_panels, "typical", invert=False)
        building.add_wall("left", [typical_panel] * building.depth_panels, "typical", invert=False)
        building.complete_walls("typical")  # Автозавершение для типового этажа

    # === ШАГ 5: КАСТОМНАЯ КРЫША (если roof_type="custom") ===
    if building.roof_type == "custom":
        # RoofGeometry.create_custom_roof() аргументы:
        #   vertices: List[Tuple[float, float, float]] - вершины в АБСОЛЮТНЫХ координатах (метры)
        #   faces: List[List[int]] - грани как индексы вершин (например, [0,1,2,3] для четырехугольника)
        #   colors: List[str] - цвета для каждой грани (в #RRGGBB)
        #   invert_flags: Optional[List[bool]] - инвертировать нормали для каждой грани (True = развернуть)
        # 
        # КООРДИНАТЫ:
        #   X: 0 -> building.width (ширина здания)
        #   Y: 0 -> roof_height (высота крыши)
        #   Z: 0 -> building.depth (глубина здания)
        roof_vertices = [
            # Основание крыши (уровень верха здания)
            (0, 0, 0),
            (building.width, 0, 0),
            (building.width, 0, building.depth),
            (0, 0, building.depth),
            # Верх бортика
            (0.2, 0.5, 0.2),
            (building.width - 0.2, 0.5, 0.2),
            (building.width - 0.2, 0.5, building.depth - 0.2),
            (0.2, 0.5, building.depth - 0.2),
            # Вывеска над входом (Z < 0 - выступает за пределы здания)
            (building.width * 0.2, 0.6, -0.3),  # Левый-нижний вывески
            (building.width * 0.8, 0.6, -0.3),  # Правый-нижний
            (building.width * 0.8, 0.9, -0.3),  # Правый-верхний
            (building.width * 0.2, 0.9, -0.3),  # Левый-верхний
            (building.width * 0.2, 0.6, 0),     # Левый-нижний задней части
            (building.width * 0.8, 0.6, 0),
            (building.width * 0.8, 0.9, 0),
            (building.width * 0.2, 0.9, 0),
        ]
        
        # Грани (индексы из roof_vertices)
        roof_faces = [
            [4, 5, 6, 7],    # Верхняя плоскость крыши
            [0, 1, 5, 4],    # Передний бортик
            [1, 2, 6, 5],    # Правый бортик
            [2, 3, 7, 6],    # Задний бортик
            [3, 0, 4, 7],    # Левый бортик
            [8, 9, 10, 11],  # Нижняя часть вывески
            [11, 10, 14, 15],# Задняя часть вывески (инвертировать!)
            [12, 13, 14, 15],# Верхняя часть вывески
            [8, 12, 15, 11], # Левая часть вывески
            [9, 13, 14, 10], # Правая часть вывески
        ]
        
        # Цвета для каждой грани
        roof_colors = [
            "#333333",  # Верх крыши
            "#444444", "#444444", "#444444", "#444444",  # Бортики
            "#AA0000",  # Нижняя часть вывески (красная)
            "#880000",  # Задняя часть
            "#660000",  # Верхняя часть
            "#990000", "#990000"  # Боковые части
        ]
        
        # Флаги инверсии нормалей: True для граней, смотрящих внутрь
        invert_flags = [
            False,  # Верх крыши (смотрит вверх)
            False, False, False, False,  # Бортики (смотрят наружу)
            False,  # Нижняя часть вывески
            True,   # Задняя часть вывески (должна смотреть НАРУЖУ от здания)
            False,  # Верхняя часть
            False, False  # Боковые части
        ]
        
        building.add_custom_roof(
            vertices=roof_vertices,
            faces=roof_faces,
            colors=roof_colors,
            invert_flags=invert_flags  # Критически важно для правильного освещения!
        )

    # === ШАГ 6: ФИНАЛИЗАЦИЯ И СОХРАНЕНИЕ ===
    # building.finalize() делает:
    #   - Автоматически создает стандартную крышу (если roof_type не "custom")
    #   - Исправляет нормали всех граней (автоматически определяет внутренние/внешние)
    #   - Вычисляет центры секций
    building.finalize()
    
    # Сохранение в файл
    filename = "<название_здания>.json"
    building.save_to_file(filename)
    print(f"✅ Здание сохранено в {filename}")
    return filename

# Запуск генерации при выполнении скрипта
if __name__ == "__main__":
    generate_<название_здания>()
```

#### ⚠️ КРИТИЧЕСКИ ВАЖНЫЕ ПРАВИЛА:
1. **Для "back" и "right" стен ВСЕГДА invert=True** (кроме случаев кастомной геометрии).
2. **Для окон жилых домов** используй `glass_color=Building.get_window_color` — это гарантирует разнообразие.
3. **Координаты вывесок** могут иметь Z < 0 (выступают за пределы здания), но Y всегда > 0.
4. **invert_flags для кастомной крыши**:
   - `True` если грань "смотрит внутрь" здания
   - `False` если "смотрит наружу"
   - Сомневаешься? Поставь `True` для задних граней вывесок/крыш.
5. **Все размеры в метрах** — никаких относительных единиц вне [0..1] в геометрических компонентах.

#### 💡 ПРИМЕР КОММЕНТАРИЯ ДЛЯ add_wall():
```python
# building.add_wall() детальное описание:
#   wall_type: 
#       "front" - стена у входа (Z=0), нормали +Z
#       "back"  - задняя стена (Z=depth), нормали -Z → invert=True
#       "left"  - левая стена (X=0), нормали -X
#       "right" - правая стена (X=width), нормали +X → invert=True
#   panels:
#       Список панелей для заполнения стены. Если panels_count > len(panels) — циклическое повторение.
#   floor_type:
#       "ground" - первый этаж (может иметь особые элементы: вход, магазины)
#       "typical" - этажи 2+ (обычно одинаковые панели)
```

#### 🚫 ЧЕГО НЕЛЬЗЯ:
- Забывать `building.finalize()` перед сохранением
- Использовать отрицательные координаты X/Y (только Z может быть <0 для вывесок)
- Не указывать `invert_flags` для кастомной крыши
- Использовать `random` вместо `Building.get_window_color`

#### 📌 СОВЕТ ДЛЯ LLM:
Когда описываешь аргументы в комментариях — пиши так, будто объясняешь новичку, который впервые видит фреймворк. Всегда указывай:
- Формат значений (например, "#RRGGBB")
- Диапазоны (например, "margin: 0.0-0.5")
- Специальные правила (например, "для back/right стен invert=True")
- Физический смысл (например, "roof_height - высота конька от основания крыши")
```

**Запрос пользователя:**  
«Создай функцию для генерации пекарни с круглым окном и вывеской "ХЛЕБ"»

**LLM вернет ТОЛЬКО ЭТОТ КОД (с комментариями):**
```python
# Импорты — только необходимые из фреймворка
from createBuilding import (
    Building, GeometrySegment, RectangleWithCutout, SimplePolygon,
    RoofGeometry
)

def generate_bakery():
    """
    Генерирует 3D-модель пекарни с круглым окном и вывеской "ХЛЕБ".
    Здание: 1 этаж, 4 панели по фасаду, коричневые стены, круглое окно слева, вывеска над входом.
    """

    # === ШАГ 1: ИНИЦИАЛИЗАЦИЯ ЗДАНИЯ ===
    # Building.__init__ аргументы:
    #   name: str - уникальное имя модели (используется в метаданных)
    #   description: str - человекочитаемое описание здания
    #   author: str = "UrbanSim3D" - автор модели
    #   version: str = "1.0" - версия модели
    #   floor_count: int - общее количество этажей (включая цоколь)
    #   width_panels: int - количество ПАНЕЛЕЙ по ширине фасада
    #   depth_panels: int - количество ПАНЕЛЕЙ по глубине здания
    #   panel_width: float - ширина ОДНОЙ панели в метрах
    #   panel_depth: float - глубина ОДНОЙ панели в метрах
    #   floor_height: float - стандартная высота типового этажа
    #   ground_floor_height: Optional[float] - высота первого этажа (если None - равна floor_height)
    #   roof_type: str - тип крыши: "simple" (двускатная), "flat" (плоская с бортиком), "custom" (ручная настройка)
    #   roof_height: Optional[float] - высота крыши от основания (если None - 0.43 * floor_height)
    building = Building(
        name="bakery_shop",
        description="Пекарня с круглым окном и вывеской ХЛЕБ",
        floor_count=1,
        width_panels=4,         # 4 панели по фасаду
        depth_panels=2,         # 2 панели по глубине
        panel_width=2.5,        # Ширина панели 2.5м
        panel_depth=3.0,        # Глубина панели 3.0м
        floor_height=4.0,       # Высокий первый этаж для витрин
        ground_floor_height=4.0,
        roof_type="custom",     # Нужна вывеска → кастомная крыша
        roof_height=1.2         # Низкая крыша для пекарни
    )

    # === ШАГ 2: СОЗДАНИЕ ГЕОМЕТРИЧЕСКИХ ПАНЕЛЕЙ ===
    # Building.create_window_frame() аргументы:
    #   margin: float (0.0-0.5) - отступ от краев панели до окна (в относительных координатах [0..1])
    #   frame_color: str - цвет рамки окна в формате #RRGGBB
    #   glass_color: Optional[str] - цвет стекла (если None - случайный темный цвет через get_window_color)
    #   invert_frame: bool - инвертировать нормали рамки (True для задних/боковых стен)
    #   invert_glass: bool - инвертировать нормали стекла
    # 
    # Особая панель с круглым окном (слева на фасаде)
    circular_window_panel = GeometrySegment()
    
    # Добавляем рамку с КРУГЛЫМ вырезом (аппроксимация 8 точками)
    circular_window_panel.add_child(RectangleWithCutout(
        cutout_points=[
            (0.35, 0.5),  # Верх
            (0.4, 0.6),   # Верх-право
            (0.5, 0.65),  # Право
            (0.6, 0.6),   # Низ-право
            (0.65, 0.5),  # Низ
            (0.6, 0.4),   # Низ-лево
            (0.5, 0.35),  # Лево
            (0.4, 0.4)    # Верх-лево
        ],
        color="#8B4513",  # Коричневая рамка
        invert=False
    ))
    
    # Добавляем стекло для круглого окна
    circular_window_panel.add_child(SimplePolygon(
        points=[
            (0.35, 0.5), (0.4, 0.6), (0.5, 0.65),
            (0.6, 0.6), (0.65, 0.5), (0.6, 0.4),
            (0.5, 0.35), (0.4, 0.4)
        ],
        color="#87CEEB",  # Голубоватое стекло
        invert=False
    ))
    
    # Стандартная панель с прямоугольным окном для остальных мест
    standard_panel = Building.create_window_frame(
        margin=0.2,
        frame_color="#8B4513",
        glass_color="#A9A9A9",  # Светло-серое стекло
        invert_frame=False,
        invert_glass=False
    )
    
    # Глухая панель для задней стены
    solid_panel = Building.create_solid_panel(
        color="#A0522D",  # Темно-коричневая
        invert=True       # Для задней стены
    )

    # === ШАГ 3: ДОБАВЛЕНИЕ СТЕН ===
    # building.add_wall() аргументы:
    #   wall_type: str - тип стены: "front" (X+), "back" (X-), "left" (Z+), "right" (Z-)
    #   panels: List[GeometryComponent] - список панелей для этой стены
    #   floor_type: str - "ground" (первый этаж) или "typical" (этажи 2+)
    #   panels_count: Optional[int] - сколько панелей разместить (если None - len(panels))
    #   invert: bool - глобальная инверсия нормалей для ВСЕЙ стены
    # 
    # ВАЖНО: Для "back" и "right" стен обычно invert=True
    # Фасад: [круглое окно, стандартное, стандартное, стандартное]
    front_panels = [circular_window_panel, standard_panel, standard_panel, standard_panel]
    building.add_wall(
        wall_type="front",       # Передняя стена (Z=0)
        panels=front_panels,     # Список из 4 панелей
        floor_type="ground",
        panels_count=4,          # Точно 4 панели
        invert=False             # Нормали смотрят наружу (+Z)
    )
    
    # Задняя стена — все глухие панели
    building.add_wall(
        wall_type="back",
        panels=[solid_panel] * building.width_panels,
        floor_type="ground",
        panels_count=building.width_panels,
        invert=True              # Инверсия для задней стены (-Z)
    )
    
    # Боковые стены — стандартные окна
    side_panel = Building.create_window_frame(
        margin=0.25,
        frame_color="#8B4513",
        glass_color="#A9A9A9",
        invert_frame=True,       # Инверсия для боковых стен
        invert_glass=False
    )
    building.add_wall(
        wall_type="left",        # Левая стена (X=0)
        panels=[side_panel] * building.depth_panels,
        floor_type="ground",
        panels_count=building.depth_panels,
        invert=False             # Нормали смотрят -X (лево)
    )
    
    building.complete_walls("ground")  # Автоматически добавит правую стену

    # === ШАГ 4: КАСТОМНАЯ КРЫША С ВЫВЕСКОЙ ===
    # RoofGeometry.create_custom_roof() аргументы:
    #   vertices: List[Tuple[float, float, float]] - вершины в АБСОЛЮТНЫХ координатах (метры)
    #   faces: List[List[int]] - грани как индексы вершин (например, [0,1,2,3] для четырехугольника)
    #   colors: List[str] - цвета для каждой грани (в #RRGGBB)
    #   invert_flags: Optional[List[bool]] - инвертировать нормали для каждой грани (True = развернуть)
    # 
    # КООРДИНАТЫ:
    #   X: 0 -> building.width (ширина здания)
    #   Y: 0 -> roof_height (высота крыши)
    #   Z: 0 -> building.depth (глубина здания)
    roof_vertices = [
        # Основание крыши
        (0, 0, 0),
        (building.width, 0, 0),
        (building.width, 0, building.depth),
        (0, 0, building.depth),
        # Верх плоской крыши
        (0.3, 0.8, 0.3),
        (building.width - 0.3, 0.8, 0.3),
        (building.width - 0.3, 0.8, building.depth - 0.3),
        (0.3, 0.8, building.depth - 0.3),
        # Вывеска "ХЛЕБ" (Z < 0 - выступает вперед)
        (building.width * 0.2, 0.9, -0.4),   # Левый-нижний
        (building.width * 0.8, 0.9, -0.4),   # Правый-нижний
        (building.width * 0.8, 1.3, -0.4),   # Правый-верхний
        (building.width * 0.2, 1.3, -0.4),   # Левый-верхний
    ]
    
    # Грани крыши и вывески
    roof_faces = [
        [4, 5, 6, 7],          # Верх крыши
        [0, 1, 5, 4],          # Передний бортик
        [1, 2, 6, 5],          # Правый бортик
        [2, 3, 7, 6],          # Задний бортик
        [3, 0, 4, 7],          # Левый бортик
        [8, 9, 10, 11],        # Вывеска (передняя грань)
    ]
    
    # Цвета
    roof_colors = [
        "#5D4037",  # Верх крыши (коричневый)
        "#6D4C41", "#6D4C41", "#6D4C41", "#6D4C41",  # Бортики (темнее)
        "#D32F2F"   # Вывеска красная (#D32F2F = Material Red 700)
    ]
    
    # Инверсия нормалей: только для задних граней (здесь их нет)
    invert_flags = [False] * 6
    
    building.add_custom_roof(
        vertices=roof_vertices,
        faces=roof_faces,
        colors=roof_colors,
        invert_flags=invert_flags
    )

    # === ШАГ 5: ФИНАЛИЗАЦИЯ И СОХРАНЕНИЕ ===
    # building.finalize() делает:
    #   - Автоматически создает стандартную крышу (если roof_type не "custom")
    #   - Исправляет нормали всех граней (автоматически определяет внутренние/внешние)
    #   - Вычисляет центры секций
    building.finalize()
    
    # Сохранение в файл
    filename = "/drive_d/Documents/CG_curs/program/buildings/bakery_shop.json"
    building.save_to_file(filename)
    print(f"✅ Здание сохранено в {filename}")
    return filename

    generate_bakery()
